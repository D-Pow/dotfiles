topath () {
    readlink -m "$1"
}

towindowspath () {
    path=$(topath "$1")
    echo $path | sed -e "s@/mnt/@@" -e "s@/@:/@"
}

cmd () {
    # For some reason, flags aren't picked up in $@, $2, etc. so just parse out the command
    commandToRun="$1"
    rest=${@/$commandToRun/""}
    /mnt/c/Windows/System32/cmd.exe "/C" "$commandToRun" $rest
}

clip () {
    echo "$1" | cmd clip
}
